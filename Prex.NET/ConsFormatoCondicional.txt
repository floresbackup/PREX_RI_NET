Public Function ConsFormatoCondicional(ByVal sParametros As String) As Boolean


   If Not bFmtEmb Then
      
      Dim nMaxNivel  As Long
      Dim oFmtC      As JSFmtCondition
      
      nMaxNivel = Val(sParametros)
   
      Grid.FormatStyles.Clear
   
      Set oFmtC = Grid.FmtConditions.Add(nMaxNivel, jgexEqual, 1)
      oFmtC.FormatStyle.FontBold = True
      oFmtC.FormatStyle.BackColor = &HE6E6E6
      Set oFmtC = Nothing
      ConsFormatoCondicional = True
      
   Else

      Dim oCol       As ItemsDetSys
      Dim oFmt       As JSFormatStyle
      Dim bExiste    As Boolean
      
      For Each oCol In oDetSys
      
         If UCase(Right(Trim(oCol.Campo), 8)) = "NIVELTAB" Then
            
            ObtenerFormatoColumna oCol.Format
                  
            If tForCol.Condicion = "" Then
            
               oCol.Format = ";15264748;-2147483630;Tahoma;1;0;0;8;0;1"
               
               ObtenerFormatoColumna oCol.Format
                              
               For Each oFmt In Grid.FormatStyles
                  If oCol.Key = oFmt.Name Then
                     bExiste = True
                     Exit For
                  End If
                  DoEvents
               Next
               
               If Not bExiste Then
                  
                  Set oFmt = Grid.FormatStyles.Add(oCol.Key)
                  
                  If tForCol.BackColor <> -2147483643 Then oFmt.BackColor = tForCol.BackColor
                  If tForCol.ForeColor <> -2147483630 Then oFmt.ForeColor = tForCol.ForeColor
                  If tForCol.FontName <> "Tahoma" Then oFmt.FontName = tForCol.FontName
                  If tForCol.FontBold Then oFmt.FontBold = tForCol.FontBold
                  If tForCol.FontUnderline Then oFmt.FontUnderline = tForCol.FontUnderline
                  If tForCol.FontStrikethru Then oFmt.FontStrikethru = tForCol.FontStrikethru
                  If tForCol.FontSize <> 8 Then oFmt.FontSize = tForCol.FontSize
                  
                  Set oFmt = Nothing

               End If
            End If
         
         End If
         
         DoEvents
      Next

   End If

End Function
